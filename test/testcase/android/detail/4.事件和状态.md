## TC-A13: 事件索引验证

### 测试目的
验证安卓端能否正确获取和显示事件索引

### 测试步骤
1. 初始化调试模式
2. 获取 state.chapterProgress.eventIndex
3. 推进剧情，验证索引递增

### 预期结果
- ✅ eventIndex 初始值为 1 或 null
- ✅ 每次章节内容事件后索引递增
- ✅ 开场白不占用索引（eventIndex 为 null）
- ✅ 结局事件后索引继续递增

### 安卓端数据模型
```kotlin
data class ChapterProgress(
  @SerializedName("eventIndex") val eventIndex: Int? = null,
  @SerializedName("eventKind") val eventKind: String = "",
  @SerializedName("eventSummary") val eventSummary: String = "",
  @SerializedName("eventFacts") val eventFacts: JsonElement? = null,
)
```

### 状态解析
```kotlin
fun parseChapterProgress(state: JsonElement): ChapterProgress? {

  return ChapterProgress(

    eventKind = progressObj.get("eventKind")?.asString ?: "",
    eventSummary = progressObj.get("eventSummary")?.asString ?: "",
```

- ✅ 结局判定: eventKind = "ending"

| eventKind | eventType | 说明 |
|-----------|-----------|------|
| opening | on_opening | 开场白 |

| ending | on_ending_success | 成功结局 |
| ending | on_ending_failure | 失败结局 |

```kotlin
data class TurnState(
  @SerializedName("canPlayerSpeak") val canPlayerSpeak: Boolean = false,
  @SerializedName("expectedRoleType") val expectedRoleType: String = "",
  @SerializedName("expectedRole") val expectedRole: String = "",
  @SerializedName("lastSpeakerRoleType") val lastSpeakerRoleType: String = "",
  @SerializedName("lastSpeaker") val lastSpeaker: String = "",
)

fun parseTurnState(state: JsonElement): TurnState? {
  val stateObj = state.asJsonObject
  val turnObj = stateObj.get("turnState")?.asJsonObject ?: return null

  return TurnState(
    canPlayerSpeak = turnObj.get("canPlayerSpeak")?.asBoolean ?: false,
    expectedRoleType = turnObj.get("expectedRoleType")?.asString ?: "",
    expectedRole = turnObj.get("expectedRole")?.asString ?: "",
    lastSpeakerRoleType = turnObj.get("lastSpeakerRoleType")?.asString ?: "",
    lastSpeaker = turnObj.get("lastSpeaker")?.asString ?: "",
  )
}
```

## 安卓端状态管理最佳实践

### 状态序列化
```kotlin
// 保存状态到本地
suspend fun saveState(state: JsonElement) {
  val stateJson = gson.toJson(state)
  settingsStore.saveGameState(stateJson)
}

// 加载状态
suspend fun loadState(): JsonElement? {
  val stateJson = settingsStore.loadGameState() ?: return null
  return gson.fromJson(stateJson, JsonElement::class.java)
}
```

### 状态更新通知
```kotlin
class GameViewModel : ViewModel() {
  private val _state = MutableStateFlow<JsonElement?>(null)
  val state: StateFlow<JsonElement?> = _state

  private val _chapterProgress = MutableStateFlow<ChapterProgress?>(null)
  val chapterProgress: StateFlow<ChapterProgress?> = _chapterProgress

  private val _turnState = MutableStateFlow<TurnState?>(null)
  val turnState: StateFlow<TurnState?> = _turnState

  fun updateState(newState: JsonElement) {
    _state.value = newState
    _chapterProgress.value = parseChapterProgress(newState)
    _turnState.value = parseTurnState(newState)
  }
}
```

### UI 更新
```kotlin
@Composable
fun GameScreen(viewModel: GameViewModel) {
  val chapterProgress by viewModel.chapterProgress.collectAsState()
  val turnState by viewModel.turnState.collectAsState()

  Column {
    // 显示事件进度
    chapterProgress?.let { progress ->
      Text("事件 ${progress.eventIndex}: ${progress.eventKind}")
      Text(progress.eventSummary)
    }

    // 显示轮次信息
    turnState?.let { turn ->
      if (turn.canPlayerSpeak) {
        TextInput(onSubmit = { content ->
          viewModel.performStep(content)
        })
      } else {
        Text("等待 ${turn.expectedRole} 发言...")
      }
    }
  }
}
```
